extern crate webserver;
use webserver::ThreadPool;
use std::env;
use std::fs::File;
use std::io::prelude::*;
use std::net::TcpStream;
use std::net::TcpListener;

// Listen for requests on localhost, port 8000, and create a new job for each incoming stream.
fn main() {
    let args: Vec<_> = env::args().collect();

    let address = env::args().nth(1).expect("USAGE: ./program [address]");

    let listener = TcpListener::bind(address).unwrap();

    let pool = ThreadPool::new(8);

    for stream in listener.incoming() {
        match stream {
            Ok(stream) => {
                pool.execute(|| {
                    handle_connection(stream);
                })
            }
            Err(_e) => {
                panic!("Connection failed");
            }
        }
    }

    println!("Shutting down.");
}

// Decide how to respond, and what HTML to render, to the contents of the TcpStream.
fn handle_connection(mut stream: TcpStream) {
    let mut buffer = [0; 512];
    stream.read(&mut buffer).unwrap();

    let get1 = b"GET / HTTP/1.1\r\n";
    let get2 = b"GET /assets/style.css HTTP/1.1\r\n";
    let get3 = b"GET /assets/script.js HTTP/1.1\r\n";

    let (status_line, filename) = if buffer.starts_with(get1) {
        ("HTTP/1.1 200 OK\r\nContent-Type: text/html\r\n\r\n", "index.html")
    } else if buffer.starts_with(get2) {
        ("HTTP/1.1 200 OK\r\nContent-Type: text/css\r\n\r\n", "assets/style.css")
    } else if buffer.starts_with(get3) {
        ("HTTP/1.1 200 OK\r\nContent-Type: text/javascript\r\n\r\n", "assets/script.js")
    } else {
        ("HTTP/1.1 404 NOT FOUND\r\nContent-Type: text/html\r\n\r\n", "404.html")
    };

    let file = File::open(filename);
    let mut file = match file {
        Ok(file) => file,
        Err(_error) => {
            panic!("Could not open the file")
        }
    };
    let mut contents = String::new();
    file.read_to_string(&mut contents).unwrap();

    let response = format!("{}{}", status_line, contents);

    stream.write(response.as_bytes()).unwrap();
    stream.flush().unwrap(); 
}
